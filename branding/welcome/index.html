Quick Commands

# Check your agent status
$ systemctl status agentos-gateway
# Pair a messaging channel
$ sudo agentos-pair add telegram
# View recent audit events
$ agentos-audit --summary
# Manage credentials
$ sudo agentos-vault list
# Follow gateway logs
$ journalctl -u agentos-gateway -f
